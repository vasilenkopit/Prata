#include <stdio.h>
int main(void)                 /* находит сумму первых 20 целых чисел */
{
    int count, sum;            /* оператор объ€влени€     */

    count = 0;                 /* оператор присваивани€   */
    sum = 0;                   /* оператор присваивани€   */
    while (count++ < 20)       /* оператор                */

    return 0;                  /* оператор возврата       */
